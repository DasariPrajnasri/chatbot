greeting_keywords = ["hello", "hi", "hey", "good morning", "good evening"]
greeting_responses = ["Hello!", "Hi there!", "Hey! How can I help?", "Greetings!"]

def chatbot_reply(user_input):
    message = user_input.lower()

    # Greeting check
    for word in greeting_keywords:
        if word in message:
            return random.choice(greeting_responses)

    # Identity questions
    if "your name" in message or "who are you" in message:
        return f"I'm {bot_name}, your AI chatbot powered by Python!"

    # Wellbeing questions
    if "how are you" in message:
        return "I'm a bot, so I don't have feelings, but I'm working flawlessly!"

    # Random number request
    if "number" in message or "random" in message:
        rand_num = random.randint(1, 200)
        return f"Hereâ€™s a random number for you: {rand_num}"

    # Password-related fun response (since string module is imported)
    if "password" in message:
        sample_pw = ''.join(random.choice(string.ascii_letters + string.digits) for _ in range(8))
        return f"Need password inspiration? Hereâ€™s one: {sample_pw}"

    # Exit check
    if message in ["bye", "exit", "quit"]:
        return "Goodbye! Have an awesome day!"

    # Fallback responses
    return random.choice([
        "Thatâ€™s fascinating!",
        "Can you explain more?",
        "Hmm, let me think...",
        "Iâ€™m still learning, but go on!"
    ])

# Main chatbot loop
